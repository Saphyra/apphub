function idozit(){
	setTimeout(function(){document.getElementById("eredmeny").innerHTML = "Lejárt az idő"}, 5000);

	document.getElementById("eredmeny").innerHTML = "Feladat időzítve.";
}

    <BUTTON onclick="idozit()">Óra indítása</BUTTON>

function idozit(){
	setInterval(function(){document.getElementById("eredmeny").innerHTML = new Date()}, 1000);
}

    <BUTTON onclick="idozit()">Óra indítása</BUTTON>
    <BUTTON onclick="megallit()">Óra megállítása</BUTTON>

let azonosito;

function idozit(){
	azonosito = setInterval(function(){document.getElementById("eredmeny").innerHTML = new Date()}, 1000);
}

function megallit(){
	clearInterval(azonosito);
}
